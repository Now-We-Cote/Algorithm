## 장난꾸러기 현수 문제풀이

```js
function solution(N, arr) {
  let result = [];
  let isFind = false;

  for (let i = 0; i < N; i++) {
    if (result.length === 3) {
      break;
    }

    if (!isFind) {
      if (arr[i] > arr[i + 1]) {
        result.push(i + 1);
        isFind = true;
      }
    } else {
      if (arr[i] < arr[i - 1]) {
        result.push(i + 1);
      }
    }
  }

  return [result[0], result[result.length - 1]];
}

solution(9, [120, 125, 152, 130, 135, 135, 143, 127, 160]);
```

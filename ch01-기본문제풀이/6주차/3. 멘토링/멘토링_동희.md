# 멘토링

```javascript
function solution(arr) {
  let answer = 0;
  const test = arr.length; //테스트 횟수
  const student = arr[0].length; //학생수
  for (let i = 1; i <= test; i++) {
    let count = 0;
    for (let j = 1; j <= student; j++) {
      for (let k = 0; k < test; k++) {
        //테스트
        let rankI = 0;
        let rankJ = 0;
        for (let s = 0; s < student; s++) {
          //각 테스트별 학생
          if (arr[k][s] === i) {
            rankI = i;
          }
          if (arr[k][s] === j) {
            rankJ = j;
          }
        }
        if (rankI < rankJ) {
          count++;
        }
      }
      if (count === test) {
        answer++;
      }
    }
  }

  return answer;
}

let arr = [
  [3, 4, 1, 2],
  [4, 3, 2, 1],
  [3, 1, 4, 2],
];
console.log(solution(arr));
```

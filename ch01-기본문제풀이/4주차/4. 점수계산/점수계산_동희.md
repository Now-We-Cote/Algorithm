# 점수계산

## 풀이

```javascript
function solution(arr) {
  let rankArr = [];
  let copyArr = [...arr];

  //오름차순으로 점수 정렬
  let sortedArr = copyArr.sort((a, b) => {
    return b - a;
  });

  //오름차순 정렬 배열의 중복제거
  let setArr = new Set(sortedArr);
  let sortedSetArr = [...setArr];

  //주어진 배열에서 등수 구하기
  for (x of arr) {
    rankArr.push(sortedSetArr.indexOf(x) + 1);
  }
  return rankArr;
}

let arr = [87, 89, 92, 100, 76];
solution(arr); //[4, 3, 2, 1, 5];
```

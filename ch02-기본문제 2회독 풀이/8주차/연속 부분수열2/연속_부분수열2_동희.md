# 연속 부분수열 2

```javascript
const solution = (n,m,arr) => {
  let start = cnt = total = 0;

  for(let end=0; end<n; end++){
    total += arr[end];
      while(total<=m){
      total -=arr[start];
      start++;
    }
  cnt += end - start + 1;
  }
 return cnt;
}

//예제
solution(5,5,[1,3,1,2,3]); //-15..?
```

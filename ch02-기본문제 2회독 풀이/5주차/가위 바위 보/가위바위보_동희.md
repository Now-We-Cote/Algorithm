const solution = (a,b) => {
    const total = a.length;
    const answer = [];
    for(let i=0; i<total; i++){
        const diff = a[i]-b[i];
        diff === 0 && answer.push("D"); //비김
        diff === 1 && answer.push("A"); //A승리
        (diff < 0 || diff >1) && answer.push("B"); //B승리
    }
    return answer;
}


let a=[2, 3, 3, 1, 3];
let b=[1, 1, 2, 2, 3];
solution(a, b); //[ 'A', 'B', 'A', 'B', 'D' ]
